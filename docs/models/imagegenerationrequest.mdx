## Example Usage

import { ImageGenerationRequest } from "@openrouter/sdk/models";

let value: ImageGenerationRequest = {
  model: "bytedance-seed/seedream-4.5",
  prompt: "a red panda astronaut floating in space, studio lighting",
};
