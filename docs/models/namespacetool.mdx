Groups function/custom tools under a shared namespace

## Example Usage

```typescript
import { NamespaceTool } from "@openrouter/sdk/models";

let value: NamespaceTool = {
  description: "Tools for spawning and managing sub-agents.",
  name: "multi_agent_v1",
  tools: [
    {
      name: "spawn_agent",
      type: "function",
    },
  ],
  type: "namespace",
};
```
