## Example Usage

```typescript
import { AgentContext } from "@openrouter/sdk/models";

let value: AgentContext = {
  createdAt: new Date("2026-05-08T17:30:00Z"),
  id: "agtctx_01h9k2m3p4q5r6s7t8u9v0w1x2",
  knowledge: [
    "Always use TypeScript strict mode",
    "Run pnpm test before committing",
  ],
  mcpServers: [
    {
      url: "https://mcp.myorg.com/jira",
    },
  ],
  name: "myorg/myapp",
  object: "agent.context",
  scope: "org",
  updatedAt: new Date("2026-05-08T17:30:00Z"),
};
```
