# Provider

When multiple model providers are available, optionally indicate your routing preference.

## Example Usage

```typescript
import { Provider } from "@openrouter/sdk/models";

let value: Provider = {
  allowFallbacks: true,
  requireParameters: false,
  dataCollection: null,
  zdr: true,
  order: [
    "Google AI Studio",
  ],
  only: [
    "Lambda",
  ],
  ignore: [
    "Atoma",
  ],
  quantizations: [
    "bf16",
  ],
  sort: "price",
  maxPrice: {
    prompt: "<value>",
    completion: "<value>",
    image: "https://picsum.photos/seed/8smhy/1881/1994",
    audio: "<value>",
    request: "<value>",
  },
  experimental: null,
};
```
